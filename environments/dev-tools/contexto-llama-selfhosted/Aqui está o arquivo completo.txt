● Aqui está o arquivo completo:                                                                                                                                                           
                  
  /home/familia/stacks/ollama/docker-compose.yml                                                                                                                                          
                  
  version: "3.8"                                                                                                                                                                          
                                                                                                                                                                                          
  services:
    ollama:
      image: ollama/ollama:latest
      ports:
        - target: 11434
          published: 11434
          protocol: tcp
          mode: host
      volumes:
        - ollama_data:/root/.ollama
      networks:
        - ollama_internal
      deploy:
        mode: replicated
        replicas: 1
        placement:
          constraints: [node.role == manager]
        resources:
          limits:
            memory: 12G
          reservations:
            memory: 4G

    open-webui:
      image: ghcr.io/open-webui/open-webui:main
      ports:
        - target: 8080
          published: 3000
          protocol: tcp
          mode: host
      environment:
        - OLLAMA_BASE_URL=http://ollama:11434
        - RAG_EMBEDDING_ENGINE=ollama
        - RAG_EMBEDDING_MODEL=nomic-embed-text
        - ENABLE_RAG_WEB_SEARCH=false
      volumes:
        - openwebui_data:/app/backend/data
      networks:
        - ollama_internal
      deploy:
        mode: replicated
        replicas: 1
        placement:
          constraints: [node.role == manager]

  networks:
    ollama_internal:
      driver: overlay

  volumes:
    ollama_data:
    openwebui_data:

  Ou se preferir ler direto no terminal:

  cat ~/stacks/ollama/docker-compose.yml
